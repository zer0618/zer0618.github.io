ctrl_interface=/var/run/wpa_supplicant

network={

    ssid="eduroam"
    key_mgmt=WPA-EAP
    proto=RSN
    eap=TTLS
#    anonymous_identity="anonymous@uni-***.de"
    identity="your-uni-id@uni-***.de"
    password="*******"
    ca_path="/etc/ssl/certs/"
    phase2="auth=PAP"

然后就可以：

sudo wpa_supplicant -B -i eth1 -Dwext -c /etc/my_wpa_supplicant.conf

之后

Kopieren der Konfigurationsdatei in das richtige Verzeichnis:

cp unibn-vpn.pcf /etc/CiscoSystemsVPNClient/Profiles/

Die Zugriffsrechte sollten noch etwas angepaßt werden:

chmod 755 /etc/CiscoSystemsVPNClient

chmod 755 /etc/CiscoSystemsVPNClient/Profiles

chmod 644 /etc/CiscoSystemsVPNClient/Profiles/unibn-vpn.pcf

Auf- und Abbau der VPN-Verbindung (unter der normalen Benutzernummer):

vpnclient connect unibn-vpn

vpnclient disconnect
